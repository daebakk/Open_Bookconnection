소스 코드 

int floor;
	printf("몇층탑을 쌓을지 입력해주세요: ");
	scanf("%d", &floor);

	for (int n = 1 ; n < floor +1  ; n++ )
	{
		for (int i = 0; i < floor - n; i++)
			{
				printf(" ");
			}

		for (int j = 0; j < (2 * n - 1); j++)
			{
				printf("*");
			}

		
		printf("\n");
	}
	printf("\n");
